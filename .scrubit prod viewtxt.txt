SELECT * FROM FUZE.enum_reference where UPPER(SUB_TYPE) not like '%CREDENTIAL%' and 
    upper(NAME) not like '%KEY%'  and upper(NAME) not like '%PASSWORD%' and upper(NAME) not like '%PWD%'
    and upper(NAME) not  like '%HOST%' AND upper(NAME) not like '%URL%'
;COMMIT;
